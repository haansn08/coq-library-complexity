Instance term_substT :
  computableTime' subst (fun s _ => (5, (fun n _ => (1, (fun t _ => ( (size s) * (size s + c__eqbComp nat * 4 + 20), tt)))))).

  extract. unfold eqbTime.

  recRel_prettify2. all:unfold eqbTime; cbn [fst snd size].
